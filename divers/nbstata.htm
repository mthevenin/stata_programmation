quietly {
clear
import excel "C:\Users\thevenin_m\Downloads\CDB_TFR.xlsx", sheet("CDB_TFR") firstrow

grstyle init
grstyle set plain, compact
grstyle set color flare, n(2) reverse


* nettoyage base
replace pays = substr(pays,4,.)
drop if y1950==.

keep if inlist(pays,"Argentina", "United States of America", "Japan", "China, Taiwan Province of China", "Republic of Korea", "France", "Germany", "Italy", "Iceland" ) | inlist(pays, "Trkiye", "Israel", "Portugal")

replace pays="Turkey" if pays=="Trkiye"
replace pays="Taiwan" if pays=="China, Taiwan Province of China"


sencode pays, gen(pays50)  gsort(-y1950)
sencode pays, gen(pays22)  gsort(-y2022)
sencode pays, replace

order pays50 pays22, after(pays)

reshape long y, i(code) j(year)
order pays50, after(pays)
* tw line y year, by(pays)
}
